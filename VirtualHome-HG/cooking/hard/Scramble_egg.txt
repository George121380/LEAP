Task name: Prepare breakfast
Goal: Scramble egg by the fryingpan.
Guidance: Put the pan on the stove. Turn on the stove. Put oil in the pan. Put the egg in the pan.
Logic: k1 or k2

behavior k1():
    body:
        bind egg: item where:
            is_food_egg(egg)
        bind pan: item where:
            is_fryingpan(pan)
        bind stove: item where:
            is_stove(stove)
        
        achieve on(pan, stove)
        achieve is_on(stove)
        symbol oil_in_pan=exists oil:item: is_oil(oil) and inside(oil, pan)
        if not oil_in_pan:
            bind oil: item where:
                is_oil(oil)
            achieve inside(oil, pan)
        achieve inside(egg, pan)
