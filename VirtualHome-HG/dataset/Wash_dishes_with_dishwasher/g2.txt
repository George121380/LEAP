Task name: Wash dishes with dishwasher
Goal: Put cleaning solution in the dishwasher. Then wash the bowls and the plates with dishwasher.
Guidance: Put a cleaning solution in the dishwasher. Then put all the bowls and the plates into the dishwasher. Turn on the dishwasher.
Logic: k1 then k2

behavior k1():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        symbol has_cleaner=exists cleaner:item: is_cleaning_solution(cleaner) and inside(cleaner, dishwasher)
        if not has_cleaner:
            bind cleaner: item where:
                is_cleaning_solution(cleaner)
            achieve inside(cleaner, dishwasher)

behavior k2():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        foreach o: item:
            if is_bowl(o) or is_plate(o) and dirty(o):
                achieve inside(o, dishwasher)
        achieve is_on(dishwasher)
