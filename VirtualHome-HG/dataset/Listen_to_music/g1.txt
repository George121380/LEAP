Task name: Listen to music
Goal: Insert a CD and play music using the CD player.
Guidance: You should first put the cd in the cd player. Then turn on the cd player.
Logic: k1 or k2

behavior k1():
    body:
        bind cd_player: item where:
            is_cd_player(cd_player)
        symbol cd_inside_cdplayer=exists cd: item: inside(cd, cd_player) and is_cd(cd)
        if not cd_inside_cdplayer:
            bind cd: item where:
                is_cd(cd)
            achieve inside(cd, cd_player)
        achieve is_on(cd_player)

behavior k2():
    body:
        bind cd_player: item where:
            is_cd_player(cd_player)
        symbol cd_inside_cdplayer=exists cd: item: inside(cd, cd_player) and is_cd(cd)
        if not cd_inside_cdplayer:
            bind cd: item where:
                is_cd(cd)
            achieve on(cd, cd_player)
        achieve is_on(cd_player)
