Task name: Prepare dinner
Goal: Boil some rice by pot.
Guidance: Fill the pot with water. Put the rice into the pot. Put the pot on the stove and turn on the stove.
Logic: k1

behavior k1():
    body:
        bind rice: item where:
            is_food_rice(rice)
        bind stove: item where:
            is_stove(stove)

        symbol pot_with_water=exists pot:item: is_pot(pot) and has_water(pot)
        if not pot_with_water:
            bind pot: item where:
                is_pot(pot)
            achieve has_water(pot)
            achieve inside(rice, pot)
            achieve on(pot, stove)
            achieve is_on(stove)

        else:
            bind pot: item where:
                is_pot(pot) and has_water(pot)
            achieve inside(rice, pot)
            achieve on(pot, stove)
            achieve is_on(stove)
