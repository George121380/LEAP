Task name: Iron clothes
Goal: Iron the shirt and the skirt.
Guidance: For each piece of clothing, place it on the ironing board. Turn on the iron, then put the iron on the clothing, push and pull the iron.

Logic: (k1 then k2) or (k2 then k1)

S0_Actions: push(iron_2089),pull(iron_2089)
S1_Actions: push(iron_2161),pull(iron_2161)
S2_Actions: push(iron_2117),pull(iron_2117)

behavior k1():
    body:
        bind shirt: item where:
            is_clothes_shirt(shirt)
        bind iron: item where:
            is_iron(iron)
        bind ironing_board: item where:
            is_ironing_board(ironing_board)
        achieve on(shirt, ironing_board)
        achieve on(iron, shirt)
        achieve is_on(iron)

behavior k2():
    body:
        bind skirt: item where:
            is_clothes_skirt(skirt)
        bind iron: item where:
            is_iron(iron)
        bind ironing_board: item where:
            is_ironing_board(ironing_board)
        achieve on(skirt, ironing_board)
        achieve on(iron, skirt)
        achieve is_on(iron)
