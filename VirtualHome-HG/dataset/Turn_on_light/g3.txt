Task name: Turn on light
Goal: Make sure there is at least one light on in every room.
Guidance: Make sure there is at least one light on in every room.
Logic: k1

behavior k1():
    body:
        foreach room in (findall room:item: is_room(room)):
            symbol has_light_on=exists light:item: is_light(light) and inside(light, room) and is_on(light)
            if not has_light_on:
                foreach light in (findall light:item: is_light(light) and inside(light, room)):
                    achieve is_on(light)
