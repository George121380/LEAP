Task name: Turn on light
Goal: Turn on the lights in the bedroom. Then turn on the lights in the bathroom.
Guidance: Turn on the lights in the bedroom. Then turn on the lights in the bathroom.
Logic: k1

behavior k1():
    body:
        bind bedroom: item where:
            is_bedroom(bedroom)
        bind bathroom: item where:
            is_bathroom(bathroom)
        bind light1: item where:
            is_light(light1) and inside(light1, bedroom)
        bind light2: item where:
            is_light(light2) and inside(light2, bathroom)
        achieve is_on(light1)
        achieve is_on(light2)
