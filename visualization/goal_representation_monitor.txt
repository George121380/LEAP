 
behavior ensure_coffee_maker_is_ready(coffe_maker:item):
    body:
        achieve has_water(coffe_maker) # Ensure the coffee maker has water

behavior brew_coffee(coffe_maker:item, cup:item):
    body:
        achieve plugged(coffe_maker) # Coffee maker must be plugged in to operate
        achieve is_on(coffe_maker) # Turn on the coffee maker to start brewing
        achieve close(cup, coffe_maker) # Ensure the cup is close enough to receive coffee

behavior __goal__():
    body:
        bind coffe_maker: item where:
            is_coffe_maker(coffe_maker)
        # Select a coffee maker
        bind cup: item where:
            is_cup(cup) and close(cup, coffe_maker)
        # Select a cup close to the coffee maker
        ensure_coffee_maker_is_ready(coffe_maker)
        brew_coffee(coffe_maker, cup)
