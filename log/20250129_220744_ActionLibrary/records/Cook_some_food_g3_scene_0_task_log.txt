    # Function to check if a pot is on a stove
    symbol on_stove = exists stove: item : is_stove(stove) and on(pot, stove)
    return on_stove

behavior fill_with_water_if_not_has(pot: item):
    body:
        if not has_water(pot):
            # If the pot does not have water, add water to it
            achieve has_water(pot)

behavior start_boiling(pot: item):
    body:
        if is_on_a_stove(pot):
            # If the pot is on a stove, turn the stove on to start boiling
            bind stove: item where:
                is_stove(stove) and on(pot, stove)
            achieve_once is_on(stove)
            # Turn the stove on to achieve boiling

behavior __goal__():
    body:

        fill_with_water_if_not_has(pot)
        start_boiling(pot)
