behavior clean_vegetables(vegetable:item, sink:item):
    # Behavior to clean vegetables using a sink
    body:
        achieve_once inside(vegetable, sink)
        # Place vegetable inside the sink

        if exists faucet: item : is_faucet(faucet) and close(faucet, sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_on(faucet)
            # Turn on the faucet if its available

        wash(vegetable)
        # Wash the vegetable

        if exists faucet: item : is_faucet(faucet) and close(faucet, sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_off(faucet)
            # Turn off the faucet after washing

        # For each dirty vegetable, clean it in the sink

    body:
        achieve inside(vegetable, sink)
        # Ensure the vegetable is placed inside the sink

        if exists faucet:item: is_faucet(faucet) and close(faucet,sink):
            achieve_once is_on(faucet)
            wash(vegetable)
            achieve_once is_off(faucet)
        else:
            wash(vegetable)
