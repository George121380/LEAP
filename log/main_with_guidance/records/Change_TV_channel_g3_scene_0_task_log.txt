    # Function to check if there is a remote control on the TV stand
    symbol has_remote=exists o: item : is_remote_control(o) and on(o, tvstand)
    return has_remote

behavior locate_remote_control(tvstand:item):
    body:
        if has_remote_control(tvstand):
            achieve close_char(char, tvstand)
        else:
            foreach tvstand: item:
