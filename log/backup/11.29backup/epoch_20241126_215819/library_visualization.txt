behavior boil_egg(egg:item, pot:item, stove:item):
    body:
        assert is_food_egg(egg) # Ensure the item is a food egg
        achieve inside(egg, pot) # Place the egg inside the pot

        assert is_pot(pot) # Ensure the pot is a valid boiling container
        achieve has_water(pot) # Ensure the pot has water for boiling

        assert is_stove(stove) # Ensure the stove is available for use
        achieve on(pot, stove) # Place the pot on the stove for boiling
        achieve is_on(stove) # Turn on the stove to boil the egg
