def inside_fridge(item: item):
    # Function to check if item is inside the fridge
    symbol item_inside_fridge = exists o: item : (inside(o, item) and is_food(o))
    return item_inside_fridge


behavior throw_all_items_to_trash(fridge: item, trashcan: item):
    body:
        foreach item: item:
            if inside(item, fridge):
                achieve_once inside(item, trashcan)


behavior __goal__():
    body:
        bind fridge: item where:
            is_fridge(fridge)
        bind trashcan: item where:
            is_trashcan(trashcan)
        throw_all_items_to_trash(fridge, trashcan)
