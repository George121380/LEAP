behavior clean_item(item:item):
    body:
        if dirty(item):
            wash(item)
        achieve clean(item)

behavior boil_chicken(chicken:item, pot:item, stove:item):
    body:
        achieve_once inside(chicken, pot) # Put the chicken inside the pot
        achieve has_water(pot) # Ensure pot has water
        achieve_once closed(stove) # Assume stove needs to be closed to turn on
        achieve is_on(stove) # Turn on stove to heat it

behavior cut_ingredients(ingredient:item, plate:item):
    body:
        achieve cut(ingredient) 
        achieve_once on(ingredient, plate) # Place the cut ingredient on the plate

behavior __goal__():
    body:
        # Bind the items from available categories
        bind vegetable: item where:

        # Clean all ingredients
