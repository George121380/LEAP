behavior clean_item(item:item, sink:item):
    body:
        achieve inside(item, sink) # Ensure the item is placed inside the sink

        if exists faucet: item: is_faucet(faucet) and close(faucet, sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_on(faucet) # Turn on the faucet if its available

        wash(item) # Wash the item

        if exists faucet: item: is_faucet(faucet) and close(faucet, sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_off(faucet) # Turn off the faucet after washing
