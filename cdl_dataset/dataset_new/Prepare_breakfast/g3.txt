Task name: Prepare breakfast
Goal: Boil egg.
Guidance: Fill the pot with water. Put the pot on the stove. Turn on the stove. Put the egg in the pot.
Logic: k1

behavior k1():
    body:
        bind egg: item where:
            is_food_egg(egg)
        bind stove: item where:
            is_stove(stove)
        bind pot: item where:
            is_pot(pot)
        achieve has_water(pot)
        achieve on(pot, stove)
        achieve is_on(stove)
        achieve inside(egg, pot)
