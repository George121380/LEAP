Task name: Cook some food
Goal: Wash all the vegetables, then put them in a pot and boil them on the stove. Then cool them in a bowl.
Guidance: Wash all the vegetables. Put them in a pot. Fill the pot with water. Put the pot on the stove. Turn on the stove. Then turn off the stove. Put the vegetables in a bowl.
Logic: (k1 or k2) then k3
S0_Actions: switchon_executor(faucet_43) or switchon_executor(faucet_232)
S1_Actions: switchon_executor(faucet_20) or switchon_executor(faucet_121)
S2_Actions: switchon_executor(faucet_134) or switchon_executor(faucet_308)

behavior k1():
    body:
        bind pot: item where:
            is_pot(pot)
        bind stove: item where:
            is_stove(stove)
        foreach o: item:
            if is_food_vegetable(o):
                achieve inside(o, pot)
        achieve on(pot,stove)
        achieve is_on(stove)

behavior k2():
    body:
        bind pot: item where:

        bind stove: item where:
            is_stove(stove)
        foreach o: item:
            if is_food_vegetable(o):

behavior k3():
    body:
        bind vegetables: item where:
            is_food_vegetable(vegetables)
        bind stove: item where:
            is_stove(stove)
        achieve is_off(stove)
        symbol has_bowl=exists bowl:item: is_bowl(bowl) and inside(vegetables, bowl)
        if not has_bowl:
            bind bowl: item where:
                is_bowl(bowl)
            achieve inside(vegetables, bowl)
