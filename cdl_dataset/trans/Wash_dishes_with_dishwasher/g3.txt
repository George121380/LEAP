Task name: Wash dishes with dishwasher
Goal: Wash the bowls and plates with dishwasher. And wash forks by hand.
Logic: k1
Actions: wash_executor(fork_2080), wash_executor(fork_2081)

behavior k1():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        foreach o: item:
            if is_bowl(o) or is_plate(o) and dirty(o):
                achieve inside(o, dishwasher)
        achieve is_on(dishwasher)
