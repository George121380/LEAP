Task name: Drink
Goal: Use pot to boil some water.
Guidance: Fill the pot with water. Then put the pot on the stove and turn on the stove.
Logic: k1

behavior k1():
    body:
    
        symbol pot_with_water=exists pot:item: is_pot(pot) and has_water(pot)
        if not pot_with_water:
            bind stove: item where:
                is_stove(stove)
            bind pot: item where:
                is_pot(pot)
            achieve has_water(pot)
            achieve on(pot,stove)
            achieve is_on(stove)

        if pot_with_water:
            bind stove: item where:
                is_stove(stove)
            bind pot: item where:
                is_pot(pot) and has_water(pot)
            achieve on(pot,stove)
            achieve is_on(stove)
