Task name: Prepare dinner
Goal: Cook bacon with the frying pan.
Guidance: Put oil in the pan. Put the pan on the stove and turn on the stove. Put the bacon in the frying pan.

Logic: k1

behavior k1():
    body:
        bind bacon: item where:
            is_food_bacon(bacon)
        bind pan: item where:
            is_fryingpan(pan)
        bind oil: item where:
            is_oil(oil)
        bind stove: item where:
            is_stove(stove)
        achieve inside(bacon, pan)
        achieve inside(oil, pan)
        achieve on(pan, stove)
        achieve is_on(stove)
