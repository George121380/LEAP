Task name: Cook some food
Goal: Make a bowl of egg salad with boiled egg, lemon, and vegetables.
Guidance: Boil an egg and cut it into pieces. To boil egg, find a pot and fill it with water. Then put the pot on a stove and turn on the stove. Put the egg in the pot. After boiling, put the egg on a cutting board and cut it into pieces. Then put the egg in a bowl. Then clean a lemon and cut it into pieces. Put the lemon in the bowl. Then wash vegetables and cut them into pieces. Put the vegetables in the bowl.

Logic: k1 then k2
behavior k1():
    body:
        bind egg: item where:
            is_food_egg(egg)
        bind stove: item where:
            is_stove(stove)
        symbol pot_with_water=exists pot:item: is_pot(pot) and has_water(pot)
        if not pot_with_water:
            bind pot: item where:
                is_pot(pot)
            achieve has_water(pot)
            achieve on(pot,stove)
            achieve is_on(stove)
            achieve inside(egg, pot)
        else:
            bind pot: item where:
                is_pot(pot) and has_water(pot)
            achieve on(pot,stove)
            achieve is_on(stove)
            achieve inside(egg, pot)


behavior k2():
    body:
        bind egg: item where:
            is_food_egg(egg)
        bind lemon: item where:
            is_food_lemon(lemon)

        achieve clean(lemon)
        achieve cut(lemon)

        bind vegetables: item where:
            is_food_vegetable(vegetables)

        achieve clean(vegetables)
        achieve cut(vegetables)

        symbol egg_in_bowl=exists bowl:item: is_bowl(bowl) and inside(egg, bowl)
        if not egg_in_bowl:
            bind bowl: item where:
                is_bowl(bowl)
            achieve inside(egg, bowl)
            achieve inside(lemon, bowl)
            achieve inside(vegetables, bowl)
        else:
            bind bowl: item where:
                is_bowl(bowl) and inside(egg, bowl)
            achieve inside(lemon, bowl)
            achieve inside(vegetables, bowl)
