Task name: Prepare breakfast
Goal: Prepare a cup of hot milk with cereal inside.
Guidance: Find a cup to put the milk in. Then add cereal into the cup. Then you can hot the milk in the microwave.
Logic: k1

behavior k1():
    body:
    
        bind milk: item where:
            is_milk(milk)
        bind cereal: item where:
            is_food_cereal(cereal)
        bind microwave: item where:
            is_microwave(microwave)

        symbol cup_with_milk=exists cup:item: is_cup(cup) and inside(milk, cup)
        if not cup_with_milk:
            bind cup: item where:
                is_cup(cup)
            achieve inside(milk, cup)
            achieve inside(cereal, cup)
            achieve inside(cup, microwave)
            achieve is_on(microwave)
        else:
            bind cup: item where:
                is_cup(cup) and inside(milk, cup)
            achieve inside(cereal, cup)
            achieve inside(cup, microwave)
            achieve is_on(microwave)
