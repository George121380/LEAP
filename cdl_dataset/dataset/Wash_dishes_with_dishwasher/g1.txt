Task name: Wash dishes with dishwasher
Goal: wash all the bowls and the plates with dishwasher
Guidance: Put all the bowls and the plates into the dishwasher. Turn on the dishwasher.
Logic: k1 then k2

behavior k1():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        foreach o: item:
            if is_bowl(o) or is_plate(o):
                achieve inside(o, dishwasher)

behavior k2():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        achieve is_on(dishwasher)
