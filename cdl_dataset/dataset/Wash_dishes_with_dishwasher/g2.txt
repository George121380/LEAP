Task name: Wash dishes with dishwasher
Goal: Put dish soap in the dishwasher. Then wash the bowls and the plates with dishwasher.
Logic: k1 then k2

behavior k1():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        symbol has_soap=exists soap:item: is_soap(soap) and inside(soap, dishwasher)
        if not has_soap:
            bind soap: item where:
                is_soap(soap)
            achieve inside(soap, dishwasher)

behavior k2():
    body:
        bind dishwasher: item where:
            is_dishwasher(dishwasher)
        foreach o: item:
            if is_bowl(o) or is_plate(o) and dirty(o):
                achieve inside(o, dishwasher)
        achieve is_on(dishwasher)
