Task name: Pet cat
Goal: Prepare a bowl of water and put the bowl on the table in home office. Then find my cat take it to the table.
Logic: k1 then k2

behavior k1():
    body:
        bind home_office: item where:
            is_home_office(home_office)
        bind table: item where:
            is_table(table) and inside(table, home_office)
        symbol has_bowl=exists bowl:item: is_bowl(bowl) and has_water(bowl) and on(bowl, table)
        if not has_bowl:
            bind bowl: item where:
                is_bowl(bowl)
            achieve has_water(bowl)
            achieve on(bowl, table)
        else:
            fail()
        
behavior k2():
    body:
        bind cat: item where:
            is_cat(cat)
        bind home_office: item where:
            is_home_office(home_office)
        bind table: item where:
            is_table(table) and inside(table, home_office)
        achieve on(cat, table)
