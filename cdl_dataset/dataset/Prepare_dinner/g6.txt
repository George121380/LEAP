Task name: Prepare dinner
Goal: Put pizza and hamburger on a plate. Then hot them in the microwave. Then put them on a table in dinning room.
Guidance: Find a plate to put the pizza and hamburger on. Then put them in the microwave. Turn on the microwave. After that, put the plate on the table in the dinning room.
Logic: k1 then k2

behavior k1():
    body:
        bind pizza: item where:
            is_food_pizza(pizza)
        bind hamburger: item where:
            is_food_hamburger(hamburger)
        bind plate: item where:
            is_plate(plate)
        bind microwave: item where:
            is_microwave(microwave)
        achieve on(pizza,plate)
        achieve on(hamburger,plate)
        achieve inside(plate,microwave)
        achieve is_on(microwave)

behavior k2():
    body:
        bind plate: item where:
            is_plate(plate)
        bind dining_room: item where:
            is_dining_room(dining_room)
        symbol table_in_dinning_room_with_plate=exists table:item: (is_table(table) and inside(table, dining_room) and on(plate, table))
        if not table_in_dinning_room_with_plate:
            bind table: item where:
                is_table(table) and inside(table, dining_room)
            achieve on(plate, table)
