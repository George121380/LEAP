Task name: Prepare dinner
Goal: Fried rice with eggs.
Logic: k1 then k2

behavior k1():
    body:
        bind rice: item where:
            is_food_rice(rice)
        bind pot: item where:
            is_pot(pot)
        bind stove: item where:
            is_stove(stove)
        achieve has_water(pot)
        achieve inside(rice, pot)
        achieve on(pot, stove)
        achieve is_on(stove)

behavior k2():
    body:
        bind eggs: item where:
            is_food_eggs(eggs)
        bind pan: item where:
            is_pan(pan)
        bind oil: item where:
            is_oil(oil)
        bind stove: item where:
            is_stove(stove)
        bind rice: item where:
            is_food_rice(rice)
        achieve inside(eggs, pan)
        achieve inside(oil, pan)
        achieve inside(rice, pan)
        achieve on(pan, stove)
        achieve is_on(stove)
