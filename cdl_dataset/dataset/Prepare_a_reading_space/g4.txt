Task name: Prepare a reading space
Goal: Find the book and put it on the coffe table in home office. Then prepare a cup of hot milk, put it on the table.
Logic: k1 then k2 then k3 then k4

behavior k1():
    body:
        bind home_office: item where:
            is_home_office(home_office)
        bind table: item where:
            is_table(table) and inside(table, home_office)
        symbol has_book=exists book:item: is_book(book) and on(book, table)
        if not has_book:
            bind book: item where:
                is_book(book)
            achieve on(book, table)

behavior k2():
    body:
        bind milk: item where:
            is_milk(milk)
        symbol cup_with_milk=exists cup: item:is_cup(cup) and inside(milk,cup)
        if not cup_with_milk:
            bind cup: item where:
                is_cup(cup)
            achieve inside(milk, cup)

behavior k3():
    body:
        bind milk: item where:
            is_milk(milk)
        bind cup_with_milk: item where:
            is_cup(cup_with_milk) and inside(milk, cup_with_milk)
        bind microwave: item where:
            is_microwave(microwave)
        achieve inside(cup_with_milk, microwave)
        achieve is_on(microwave)

behavior k4():
    body:
        bind milk: item where:
            is_milk(milk)
        bind home_office: item where:
            is_home_office(home_office)
        bind table: item where:
            is_table(table) and inside(table, home_office)
        bind cup_with_milk: item where:
            is_cup(cup_with_milk) and inside(milk, cup_with_milk)
        achieve on(cup_with_milk, table)
