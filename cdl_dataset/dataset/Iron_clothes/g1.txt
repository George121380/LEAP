Task name: Iron clothes
Goal: Clean a shirt and iron this shirt.
Guidance: Wash the shirt. Put it on the ironing board. Turn on the iron. Then put the iron on the shirt, push and pull it on the shirt.

Logic: k1 then k2
Actions: push(iron_2089),pull(iron_2089)

behavior k1():
    body:
        bind shirt: item where:
            is_clothes_shirt(shirt)
        achieve clean(shirt)

behavior k2():
    body:
        bind iron: item where:
            is_iron(iron)
        bind shirt: item where:
            is_clothes_shirt(shirt)
        bind ironing_board: item where:
            is_ironing_board(ironing_board)
        achieve on(shirt, ironing_board)
        achieve on(iron, shirt)
        achieve is_on(iron)
