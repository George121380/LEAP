Task name: Iron clothes
Goal: Iron all the clothes in the basket, and place them in the closet.
Logic: k1 then k2
Actions: push(iron_2079),pull(iron_2079)

behavior k1():
    body:
        bind basket: item where:
            basket_for_clothes(basket)
        bind iron: item where:
                is_iron(iron)
        bind ironing_board: item where:
                is_ironing_board(ironing_board)
        foreach cloth in (findall c:item: is_clothes(c) and inside(c, basket)):
                achieve on(cloth, ironing_board)
                achieve is_on(iron)

behavior k2():
    body:
        bind closet: item where:
            is_closet(closet)
        foreach cloth in (findall c:item: is_clothes(c) and on(c, ironing_board)):
                achieve inside(cloth, closet)
