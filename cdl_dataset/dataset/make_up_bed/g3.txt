Task name: make up bed
Goal: Store all the clothes on the bed to the closetdrawer. Then find two pillows and put them on the bed. And prepare a cup of water on the nightstand.
Logic: k1 then k2 then k3

behavior k1():
    body:
        bind bed: item where:
            is_bed(bed)
        bind closetdrawer: item where:
            is_closetdrawer(closetdrawer)
        foreach cloth in (findall c: item: on(c, bed) and is_clothes(c)):
            achieve inside(cloth, closetdrawer)

behavior k2():
    body:
        bind bed: item where:
            is_bed(bed)
        symbol first_pillow=exists pillow:item: is_pillow(pillow) and on(pillow, bed)
        if first_pillow:
            bind pillow1: item where:
                is_pillow(pillow1) and on(pillow1, bed)
            symbol second_pillow=exists pillow2:item: is_pillow(pillow2) and on(pillow2, bed) and pillow2!=pillow1
            if not second_pillow:
                fail()
        else:
            fail()

behavior k3():
    body:
        bind nightstand: item where:
            is_nightstand(nightstand)
        bind cup: item where:
            is_cup(cup)
        achieve has_water(cup)
        achieve on(cup, nightstand)
