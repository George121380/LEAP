Task name: Cook some food
Goal: Put vegetables in a pot and boil them on the stove.
Logic: k1

behavior k1():
    body:
        bind pot: item where:
            is_pot(pot)
        bind stove: item where:
            is_stove(stove)
        foreach o: item:
            if is_vegetable(o):
                achieve inside(o, pot)
        achieve on(pot,stove)
        achieve is_on(stove)
