Task name: Cook some food
Goal: Make oil fried banana.
Guidance: The banana is already peeled. To fried it, find a fryingpan, put it on the stove. Turn on the stove and add some oil. Put the banana in the fryingpan.
Logic: k1

behavior k1():
    body:
        bind pan: item where:
            is_fryingpan(pan)
        bind stove: item where:
            is_stove(stove)
        bind banana: item where:
            is_food_banana(banana)
        bind oil: item where:
            is_oil(oil)
        achieve on(pan,stove)
        achieve is_on(stove)
        achieve inside(oil, pan)
        achieve inside(banana, pan)
