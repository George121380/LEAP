Task name: Cook some food
Goal: Wash all the vegetables, then put them in a pot and boil them on the stove. Then cool them in a bowl of water.
Logic: k1 then k2

behavior k1():
    body:
        bind pot: item where:
            is_pot(pot)
        bind stove: item where:
            is_stove(stove)
        foreach o: item:
            if is_food_vegetable(o):
                achieve inside(o, pot)
        achieve on(pot,stove)
        achieve is_on(stove)

behavior k2():
    body:
        bind vegetables: item where:
            is_vegetable(vegetables)
        bind stove: item where:
            is_stove(stove)
        achieve is_off(stove)
        symbol has_bowl=exists bowl:item: is_bowl(bowl) and inside(vegetables, bowl)
        if not has_bowl:
            fail()
