Task name: Cook some food
Goal: Wash all the vegetables, then put them in a pot and boil them on the stove. Then store the whole pot in the fridge.
Logic: k1 then k2

behavior k1():
    body:
        bind pot: item where:
            is_pot(pot)
        bind stove: item where:
            is_stove(stove)
        foreach o: item:
            if is_food_vegetable(o):
                achieve inside(o, pot)
        achieve on(pot,stove)
        achieve is_on(stove)

behavior k2():
    body:
        bind stove: item where:
            is_stove(stove)
        bind fridge: item where:
            is_fridge(fridge)
        achieve is_off(stove)
        achieve inside(pot, fridge)
