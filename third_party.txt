# Third-Party Dependencies for LEAP Agent
# 
# This project requires two external Python libraries that need to be installed manually:

## 1. Jacinle - A Python toolkit for researchers
# Repository: https://github.com/vacancy/Jacinle
# Description: Provides utilities for deep learning research and development
git clone https://github.com/vacancy/Jacinle --recursive
export PATH=<path_to_jacinle>/bin:$PATH
export PYTHONPATH=<path_to_jacinle>:$PYTHONPATH

## 2. Concepts - Concept learning and reasoning framework  
# Repository: https://github.com/vacancy/Concepts
# Description: Framework for concept learning, reasoning, and planning
git clone https://github.com/vacancy/Concepts --recursive
export PATH=<path_to_concepts>/bin:$PATH
export PYTHONPATH=<path_to_concepts>:$PYTHONPATH

# Note: The setup.sh script will automatically handle the installation and 
# environment variable configuration for these dependencies.
