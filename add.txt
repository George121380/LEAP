# Example-4:
When the goal is: I get a bucket full of soapy water. I get my mop wet with the water and scoot it around the floor until the floor is clean.
original version:
behavior clean_with_mop(floor:item):
    body:
        bind bucket: item where:
            is_mop_bucket(bucket)
        bind mop: item where:
            is_mop(mop)
        achieve inside(mop, bucket)
        achieve clean(floor)

behavior clean_with_vacuum(floor:item):
    body:
        bind vacuum: item where:
            is_vacuum_cleaner(vacuum)
        achieve holds_rh(char, vacuum)
        achieve is_on(vacuum)
        achieve clean(floor)
    eff:
        is_clean[floor] = True

behavior __goal__():
    body:
        bind floor: item where:
            is_floor(floor)
        clean_with_mop(floor)
        clean_with_vacuum(floor)
