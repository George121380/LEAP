    # Function to check if the sink has a faucet that can be used
    symbol exists_faucet = exists faucet: item : is_faucet(faucet) and close(faucet, sink)
    return exists_faucet

behavior wash_vegetable_in_sink(vegetable: item, sink: item):
    body:
        achieve inside(vegetable, sink)
        # Ensure the vegetable is placed inside the sink for washing
        
        if has_faucet(sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_on(faucet)
            # If the sink has a faucet, turn it on to allow water flow for washing
            
        wash(vegetable)
        # Wash the vegetable
        
        if has_faucet(sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_off(faucet)
            # Turn off the faucet after washing the vegetable

                is_faucet(faucet) and close(faucet, sink)

        if exists faucet: item : is_faucet(faucet) and close(faucet, sink):
            bind faucet: item where:
                is_faucet(faucet) and close(faucet, sink)
            achieve_once is_off(faucet)
            # Turn off the faucet after washing
