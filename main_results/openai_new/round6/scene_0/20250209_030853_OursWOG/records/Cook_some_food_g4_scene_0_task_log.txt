def has_vegetable_nearby():
    # Function to check if theres a suitable vegetable nearby
    symbol nearby_vegetable = exists vegetable: item : (is_food_vegetable(vegetable) and (close_char(char, vegetable) or close(vegetable, char)))
    return nearby_vegetable

behavior gather_vegetable_and_pot(vegetable:item, pot:item):
    body:
        achieve_once close_char(char, vegetable)
        # Move to the vegetable first to gather them
        achieve_once inhand(vegetable)
        # Pick up the vegetable
        achieve_once close_char(char, pot)
        # Move to the pot with vegetable in hand
        achieve_once inside(vegetable, pot)
        # Place the vegetable in the pot

behavior fill_pot_with_water(pot:item):
    body:
        achieve has_water(pot)
        # Fill the pot with water

behavior boil_vegetables(stove:item, pot:item):
    body:
        achieve_once close_char(char, stove)
        # Approach the stove
        achieve_once on(pot, stove)
        # Place the pot on the stove
        achieve plugged(stove)
        # Plug in the stove
        achieve is_on(stove)
        # Turn on the stove to boil the vegetables

behavior __goal__():
    body:
        if has_vegetable_nearby():
            bind vegetable: item where:
                is_food_vegetable(vegetable) and (close_char(char, vegetable) or close(vegetable, char))
            # Select a nearby vegetable
            bind pot: item where:
                is_pot(pot)
            # Select a pot
            bind stove: item where:
                is_stove(stove)

            gather_vegetable_and_pot(vegetable, pot)
            fill_pot_with_water(pot)
            boil_vegetables(stove, pot)

        else:
            # Explore unvisited items to locate suitable herbs or vegetables
            foreach item: item:
                if is_food_vegetable(item) and not visited(item):
                    observe(item, "Check if the vegetable is suitable for boiling")
