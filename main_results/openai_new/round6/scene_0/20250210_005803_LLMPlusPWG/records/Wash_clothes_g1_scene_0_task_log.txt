def basket_has_clothes(basket:item):
    # Function to check if the basket has clothes inside
    symbol has_clothes=exists o: item : (is_clothes(o) and inside(o, basket))
    return has_clothes

behavior put_clothes_in_washing_machine(c:item, washing_machine:item):
    body:
        achieve_once inside(c, washing_machine) 
        # Place clothes inside the washing machine

behavior close_washing_machine(washing_machine:item):
    body:
        achieve_once closed(washing_machine)
        # Close the washing machine door

behavior fill_washing_machine_with_water(washing_machine:item):
    body:
        achieve has_water(washing_machine)
        # Ensure the washing machine has water

behavior __goal__():
    body:
        bind basket: item where:
            is_basket_for_clothes(basket)
        # Select the basket for clothes
        
        bind washing_machine: item where:
            is_washing_machine(washing_machine)
        # Select the washing machine
        
        if basket_has_clothes(basket):
            foreach c: item:
                if is_clothes(c) and inside(c, basket):
                    put_clothes_in_washing_machine(c, washing_machine)
            # Load all clothes from the basket into the washing machine
        
        close_washing_machine(washing_machine)
        fill_washing_machine_with_water(washing_machine)
        achieve is_on(washing_machine)
        # Turn on the washing machine
