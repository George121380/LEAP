            wash(produce)

behavior find_and_prepare_ingredients():
    body:
        bind kitchen: item where:
            is_kitchen(kitchen)
        
        bind ingredients: listitem where:
            is_food_rice(ingredients) or is_food_egg(ingredients) or is_food_vegetable(ingredients)
        
        foreach ingredient: item:
            if not visited(ingredient):
                walk_to(ingredient)
                observe(ingredient, "Check the condition of the ingredient, whether it is clean or needs washing")
                if dirty(ingredient):
                    wash_ingredient(ingredient)

def find_and_prepare_ingredients():
    # Define a function to find and prepare necessary ingredients for fried rice with egg
    symbol found_ingredients = exists ingredient: item : (
        (is_food_rice(ingredient) or is_food_egg(ingredient) or is_food_vegetable(ingredient))
        and visited(ingredient)
    )
    return found_ingredients

behavior prepare_ingredient(ingredient: item):
    body:
        if is_food_vegetable(ingredient) and dirty(ingredient):
            walk_to(ingredient)
            wash(ingredient) # Wash dirty vegetables
        if is_food_egg(ingredient) and not cut(ingredient):
            grab(ingredient)
            cuts(ingredient) # Crack the egg and put it carefully
            empty_a_hand()

behavior wash_ingredients():
    body:
        foreach ingredient: item:
            if is_food_vegetable(ingredient) and dirty(ingredient) and not visited(ingredient):
                observe(ingredient, "Check if the vegetable is clean")
                prepare_ingredient(ingredient)

behavior prepare_eggs():
    body:
        foreach egg: item:
            if is_food_egg(egg) and not visited(egg):
                observe(egg, "Check if the egg is cracked")
                prepare_ingredient(egg)

behavior __goal__():
    body:
        if not find_and_prepare_ingredients():
            # If ingredients are not found or prepared yet, initiate the process to find and prepare them
            wash_ingredients()
            prepare_eggs()
