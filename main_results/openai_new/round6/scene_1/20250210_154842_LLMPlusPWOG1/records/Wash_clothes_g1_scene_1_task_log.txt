def is_clothes_inside_basket(basket:item):
    # Check if there is any clothing inside the basket
    symbol there_are_clothes= exists clothes: item : (is_clothes(clothes) and inside(clothes, basket))
    return there_are_clothes

            is_basket_for_clothes(basket)

        # Select the washing machine

        if is_clothes_inside_basket(basket):
            # If there are clothes in the basket, load them into the washing machine
            foreach clothes: item:
                if is_clothes(clothes) and inside(clothes, basket):

            start_washing_machine(washing_machine) # Start the washing machine
