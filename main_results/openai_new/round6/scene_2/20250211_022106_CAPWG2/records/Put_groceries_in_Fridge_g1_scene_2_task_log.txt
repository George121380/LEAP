behavior grab_items_for_table(milk:item, turkey:item, fridge:item):
    body:
        if not inhand(milk):
            grab(milk)
        if not inhand(turkey):
            grab(turkey)
        put_on_table(milk)
        empty_a_hand()
        walk_to(fridge)  # Might need to walk to fridge to grab the turkey if its faraway
        if not inhand(turkey):
            grab(turkey)
        put_on_table(turkey)
        empty_a_hand()

behavior grab_more_items_for_table(egg:item, fish:item, chicken:item, fridge:item):
    body:
        if not inhand(egg):
            grab(egg)
        if not inhand(fish):
            grab(fish)
        put_on_table(egg)
        put_on_table(fish)
        empty_a_hand()
        walk_to(fridge)  # Might need to walk to fridge to grab the chicken if its faraway
        if not inhand(chicken):
            grab(chicken)
        put_on_table(chicken)
        empty_a_hand()

behavior put_on_table(item:item):
    body:
        bind table: item where:
            is_table(table) and is_visited(table)
        # Move the item to the table in the dining room
        if not on(item, table):
            walk_to(table) 
            put_on(item, table)
