def empty_fridge(fridge: item):
    # Function to empty all items from the fridge
    foreach item: item:
        if inside(item, fridge):
            remove_from_fridge(item)

behavior remove_from_fridge(item: item):
    body:
        if inside(item, fridge):
            grab(item)
            # Since youre removing the item, assume youll release it afterward
            empty_a_hand()
