def has_milk_and_sugar_nearby(coffee:item):
    # Check if milk and sugar are near the coffee
    symbol milk_and_sugar_nearby = exists milk: item : (is_milk(milk) and close(milk, coffee)) and exists sugar: item : (is_food_sugar(sugar) and close(sugar, coffee))
    return milk_and_sugar_nearby

behavior add_milk_and_sugar(coffee:item, milk:item, sugar:item):
    # Add milk and sugar to the coffee
    body:
        achieve_once inside(milk, coffee)
        achieve_once inside(sugar, coffee)

behavior __goal__():
    body:
        bind coffee: item where:
            is_cup(coffee) and is_drinkable(coffee)
            # Assume the coffee is in a cup
        if has_milk_and_sugar_nearby(coffee):
            # If milk and sugar are near the coffee, add them directly
            bind milk: item where:
                is_milk(milk) and close(milk, coffee)
            bind sugar: item where:
                is_food_sugar(sugar) and close(sugar, coffee)
            add_milk_and_sugar(coffee, milk, sugar)

        else:
            # If milk and sugar are not nearby, explore the environment to find them
            foreach milk: item:
                if is_milk(milk) and not visited(milk):
                    observe(milk, "Check if there is milk")
                    if close(milk, coffee):
                        bind sugar: item where:
                            is_food_sugar(sugar) and close(sugar, coffee)
                        add_milk_and_sugar(coffee, milk, sugar)
