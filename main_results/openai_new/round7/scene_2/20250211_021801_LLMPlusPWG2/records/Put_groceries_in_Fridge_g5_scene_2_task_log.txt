behavior place_on_dining_table(milk:item, egg:item, dining_table:item):
    body:
        achieve_once on(milk, dining_table)
        achieve_once on(egg, dining_table)
        
behavior throw_all_in_fridge(fridge:item, trashcan:item):
    body:
        foreach o: item:
            if inside(o, fridge):
                achieve_once inside(o, trashcan)

behavior put_back_in_fridge(milk:item, egg:item, fridge:item):
    body:
        achieve_once inside(milk, fridge)
        achieve_once inside(egg, fridge)

behavior __goal__():
    body:
        # Identify dining room table
        bind dining_table: item where:
            is_table(dining_table) and exists room: item : (is_dining_room(room) and inside(dining_table, room))

        # Identify milk and egg
        bind milk: item where:
            is_milk(milk)
        bind egg: item where:
            is_food_egg(egg)

        # Identify fridge and move milk and egg to dining table
        bind fridge: item where:
            is_fridge(fridge)
            
        put_on_table(milk, dining_table)
        put_on_table(egg, dining_table)

        # Identify trashcan
        bind trashcan: item where:
            is_trashcan(trashcan)

        # Clear items from the fridge and then store milk and egg back
        foreach item_in_fridge: item:
            if inside(item_in_fridge, fridge):
                clear_fridge(item_in_fridge, trashcan)
        
        store_in_fridge(milk, fridge)
