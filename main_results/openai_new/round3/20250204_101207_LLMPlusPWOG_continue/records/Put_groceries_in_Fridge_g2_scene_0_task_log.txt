behavior wash_fish(fish:item, sink:item):
    body:
        achieve inside(fish, sink)
        # Ensure the fish is placed inside the sink

        bind faucet: item where:
            is_faucet(faucet) and close(faucet, sink)
        achieve_once is_on(faucet)
        # Turn on the faucet to wash the fish
        
        wash(fish)
        
        achieve_once is_off(faucet)
        # Turn off the faucet after washing

behavior place_on_table(items: list[item], table:item):
    body:
        foreach item: item in items:
            achieve on(item, table)
        # Place each item on the specified table

        # Select a table in the dining room

        bind milk: item where:

        bind turkey: item where:
