def is_usable_stove(stove:item):
    # Determine whether the stove can be used
    symbol usable_stove = has_switch(stove) and not is_on(stove)
    return usable_stove

behavior boil_vegetables_on_stove(vegetables:item, stove:item, pot:item):
    body:
        # Place pot with vegetables on the stove
        put_on(pot, stove)
        # Switch on the stove
        switch_on(stove)
        yield  # Assume time passes allowing vegetables to boil
        assert is_heated(vegetables)
        # Switch off the stove
        switch_off(stove)
